87      @Override
88      public void initButtons() { 
89          int verticalMargin = 220; 
90          int horizontalMargin = 450; 
91  
92          int x1 = (int) 0; 
93  
94          MenuButton []menuButtons = new MenuButton[Globals.CITY_COUNT+1];
95  
96  
97          Image backButtonImage;
98          try {backButtonImage = new Image(BACK_BUTTON_PATH);}
99          catch (SlickException e) {backButtonImage = null;}
100 
101         // Back Button
102         menuButtons[0] = new MenuButton(sbg.getContainer(), backButtonImage , 0, 0) {
103             @Override
104             public void performAction() {
105                 sbg.enterState(Globals.MAIN_MENU);
106             }
107         }
108         ;
109 
110         for (int i=1; i <= Globals.CITY_COUNT; i++){
111 
112             final boolean unlocked; 
113 
114             // image path for the image of the button
115             String imagePath = null;
116 
117             if (unlocked = (i <= Game.getInstance().getAccount().getHighestLevelReached()))
118                 imagePath = areaImagePaths[i-1]; 
119             else
120                 imagePath = lockedAreaImagePaths[i-1];
121 
122             // Position
123             // arrange the images (2x3) (row x column)
124             int x = (int) x1 +  horizontalMargin* ((i-1)%3); 
125             int y = (int)  verticalMargin * (1+ (int)((i-1)/3));
126 
127             // Image
128             Image areaImage;
129             try {areaImage = new Image (imagePath);}
130             catch (SlickException e) {areaImage = null;}
131 
132             // used to get the path for the area
133             // but also for the level of the area
134             final int index = i; 
135             menuButtons[i] =  new MenuButton(sbg.getContainer(), areaImage, x, y) {
136                 @Override
137                 public void performAction() {
138                     if (unlocked)
139                         enterPlay(index, Globals.cityTMXPaths[index-1]);
140                 }
141             }
142             ;
143         }
144 
145         this.setButtons(menuButtons);
146     }
