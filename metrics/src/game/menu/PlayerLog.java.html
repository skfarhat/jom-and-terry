153     public void draw(int timer){ 
154 
155         final Area area = Play.getInstance().getArea(); 
156 
157         if ((person instanceof Robber)){
158             Robber robber = (Robber) person; 
159             String scoreString = String.format("Score: %d", (int) robber.getScore());
160             String moneyString = String.format("$%d", robber.getMoney());
161             String timerString = String.format("%s", getTime(timer));
162 
163             g.setColor(Color.white);
164             g.drawImage(background, position.getX(), position.getY());
165             g.drawString(scoreString, position.getX() + SCORE_X, position.getY()+ SCORE_Y);
166             g.drawString(moneyString, position.getX() + MONEY_X, position.getY()+ MONEY_Y);
167             g.drawString(timerString, position.getX() + TIMER_X, position.getY()+ TIMER_Y);
168             g.drawString(
169                     String.format("Robbed: %d/%d",
170                             area.getNumberOfRobbedBuildings(),
171                             area.getBuildings().size()),
172                             position.getX() + ROBBED_BLDGS_X, position.getY() + ROBBED_BLDGS_Y);
173             g.drawString(logText,position.getX() + LOG_X, position.getY()+ LOG_Y);
174         }
175         else {
176             Policeman policeman = (Policeman) person; 
177 
178             String scoreString = String.format("Score: %d",(int) area.getPoliceOffice().getScore());
179             String timerString = String.format("%s", getTime(timer));
180 
181             g.setColor(Color.white);
182             g.drawImage(background, position.getX(), position.getY());
183             g.drawString(scoreString, position.getX() + SCORE_X, position.getY()+ SCORE_Y);
184             g.drawString(timerString, position.getX() + TIMER_X, position.getY()+ TIMER_Y);
185             g.drawString(
186                     String.format("Gathers:%d", area.getPoliceOffice().getGathersRemaining()),
187                     position.getX() + GATHERS_X, position.getY()+ GATHERS_Y);
188             g.drawString(
189                     String.format("Robbed: %d/%d",
190                             area.getNumberOfRobbedBuildings(),
191                             area.getBuildings().size()),
192                             position.getX() + ROBBED_BLDGS_X, position.getY() + ROBBED_BLDGS_Y);
193             g.drawString(logText,position.getX() + LOG_X, position.getY()+ LOG_Y);
194         }
195 
196         for (MenuButton button: buttons){
197             button.render(g);
198         }
199 
200         checkForButtonClicks();
201 
202     }

210     private String getTime(int time){
211         int mn;
212         int sec;
213 
214         int rem = (int)(time%3600);
215         mn = rem/60;
216         sec = rem%60;
217 
218         String mnStr = (mn<10 ? "0" : "")+mn;
219         String secStr = (sec<10 ? "0" : "")+sec;
220         return String.format("%s:%s", mnStr, secStr);
221 
222     }
