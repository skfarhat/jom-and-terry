1   package game.city.building;
2   
3   import org.newdawn.slick.Image;
4   import org.newdawn.slick.Renderable;
5   import org.newdawn.slick.SlickException;
6   
7   public class Flag implements Renderable{
8       
9       enum FlagType{
10          NONE, 
11          RED, 
12          DARK_BLUE, 
13          VIOLET, 
14          GREEN
15      }; 
16      
17      
18      
19      // Static fields
20      static private String FLAG_IMAGE_PATH_RED = "res/Flags/red.png";
21      static private String FLAG_IMAGE_PATH_GREEN = "res/Flags/green.png";
22      static private String FLAG_IMAGE_PATH_DARK_BLUE = "res/Flags/dark-blue.png";
23      static private String FLAG_IMAGE_PATH_VIOLET = "res/Flags/violet.png";
24      
25  //  static private int FLAGS_COUNT = 4;
26      
27      static private Image redFlag;
28      static private Image greenFlag;
29      static private Image violetFlag;
30      static private Image darkBlueFlag;
31  
32   
33      // Static block
34      static{
35          try {
36              redFlag = new Image(FLAG_IMAGE_PATH_RED);
37              greenFlag = new Image(FLAG_IMAGE_PATH_GREEN);
38              darkBlueFlag = new Image(FLAG_IMAGE_PATH_DARK_BLUE);
39              violetFlag = new Image(FLAG_IMAGE_PATH_VIOLET);
40          } catch (SlickException e) {
41              e.printStackTrace();
42          }
43      }
44      
45      // Fields
46      public int flagID = 0;
47      public FlagType flagType; 
48      protected Image image; 
49  
50      /**
51       * default CONSTRUCTOR
52       */
53      public Flag() {
54          this.flagType = FlagType.NONE;
55          this.image = null;
56      }
57  
58      public Flag(int flagID) {
59          
60          // set the flag id one less, then call nextFlag()
61          this.flagID = flagID-1;
62          
63          // this will increment the flagID back 
64          nextFlag();
65      }
66      
67      @Override
68      public void draw(float x, float y) {
69          // draw image
70          image.draw(x,y);
71      }
72  
73      protected void nextFlag(){
74          flagID++; 
75          switch (flagID) {
76          case 0: 
77              this.image = null;
78              this.flagType = FlagType.NONE;
79              break;
80          case 1:
81              this.image = redFlag;
82              this.flagType = FlagType.RED;
83              break;
84          case 2:
85              this.image = greenFlag;
86              this.flagType = FlagType.GREEN; 
87              break;
88          case 3:
89              this.image = darkBlueFlag;
90              this.flagType = FlagType.DARK_BLUE;
91              break;
92          case 4:
93              this.image = violetFlag;
94              this.flagType = FlagType.VIOLET;
95              break;
96          default:
97              this.flagID = 0; 
98              this.image = null;
99              this.flagType = FlagType.NONE;
100             break;
101         } 
102     }
103 }
