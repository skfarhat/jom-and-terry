1   package game.city.building;
2   
3   import game.Game;
4   
5   import org.newdawn.slick.Graphics;
6   import org.newdawn.slick.geom.Rectangle;
7   
8   public class BuildingInfo {
9   
10      protected final static int BUILDING_INFO_HEIGHT = 50; 
11  
12      private Rectangle frame;  
13  
14      /**
15       * Graphics instance of the container 
16       */
17      private Graphics g = Game.getInstance().getContainer().getGraphics();
18  
19      /**
20       * Building for which the information is being displayed
21       */
22      private Building bldg; 
23  
24      /**
25       * Filling bar that shows how far the building is in the process of being robbed
26       */
27      private FillingBar fillingBar;
28  
29      // can be accessed through the building field but why add a layer of indirection when it won't change
30      /**
31       * String containing the building type. Can access it through the building field
32       */
33      private String buildingType = null; 
34  
35      /**
36       * 
37       * @param bldg Building for which the information is being displayed
38       */
39      public BuildingInfo(Building bldg) {
40  
41          this.frame = new Rectangle(bldg.position.getX(), bldg.position.getY(), bldg.width, BUILDING_INFO_HEIGHT);
42  
43          // set the building field
44          this.bldg = bldg; 
45  
46          // set the building type
47          if (bldg instanceof House){
48              this.buildingType = "House";
49          }
50          else if (bldg instanceof Shop){
51              this.buildingType = "Shop";
52          }
53          else if (bldg instanceof Bank){
54              this.buildingType = "Bank";
55          }
56   
57          // initialize the filling bar 
58          fillingBar = new FillingBar(frame.getX(), frame.getY(), frame.getWidth());
59      }
60  
61      public void draw(float x, float y) {
62  
63          // draw the filling bar
64          fillingBar.draw(frame.getX(), frame.getY() - 50);
65  
66          // the type of the building
67          g.drawString(buildingType, frame.getX(), frame.getY()-35);
68  
69          g.drawString(String.format("%d/%d",bldg.occupants.size()-bldg.occupantsOnVacation, bldg.occupants.size()), frame.getX(), frame.getY()-25);
70  
71          g.drawString(String.format("$%d", bldg.money), frame.getX(), frame.getY() -12);
72  
73  
74      }
75  
76      public FillingBar getFillingBar() {
77          return fillingBar;
78      }
79  
80      public Rectangle getFrame() {
81          return frame;
82      }
83  }
