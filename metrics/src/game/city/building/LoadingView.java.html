5   import java.io.InputStream;

7   import org.newdawn.slick.Color;
8   import org.newdawn.slick.Font;
9   import org.newdawn.slick.Graphics;
10  import org.newdawn.slick.Renderable;
11  import org.newdawn.slick.TrueTypeFont;
12  import org.newdawn.slick.geom.Rectangle;
13  import org.newdawn.slick.util.ResourceLoader;
14  
15  public class LoadingView {
16  
17      private Font font; 
18      private final static String FONT_PATH = "res/Font/Petitinho.ttf";
19      private final static float FILLING_BAR_HEIGHT = 12.0f; 
20      private Rectangle filledPart; 
21      private Rectangle mainRect; 
22      private String filledPercentString = ""; 
23  
24      public LoadingView(float x, float y, float width) {
25          mainRect = new Rectangle(x, y, width, FILLING_BAR_HEIGHT);
26  
27          // initially the width for the filled part is 0 
28          filledPart = new Rectangle(x,y,0,FILLING_BAR_HEIGHT);
29  
30          InputStream inputStream = ResourceLoader.getResourceAsStream(FONT_PATH);
31          try {
32              java.awt.Font awtFont2 = java.awt.Font.createFont(java.awt.Font.TRUETYPE_FONT, inputStream);
33              awtFont2 = awtFont2.deriveFont(11.0f); // set font size
34  
35              this.font = new TrueTypeFont(awtFont2, true);
36  
37          } catch (Exception e) {
38              e.printStackTrace();
39              this.font = null; 
40          }
41      }
42  
43  
44      public void update(float percent){
45          filledPart.setWidth(percent*mainRect.getWidth());
46          filledPercentString = String.format("%d%%", (int) (percent*100));
47      }
48  
49      public void reset(){
50          update(0.0f);
51      }
52  
53      public void draw(){
54  
55          // get the Graphics instance
56          Graphics g = Game.getInstance().getContainer().getGraphics();
57          // get the initial color to be able to reset it later
58          Color intialColor = g.getColor();
59  
60          // draw the empty rect
61          g.draw(mainRect);
62  
63          // draw the filled part of the rect
64          g.fill(filledPart);
65  
66          // Draw the percentage
67          g.setFont(font); 
68          g.setColor(Color.blue);
69  //      g.drawString(filledPercentString, mainRect.getX(), mainRect.getY());
70  
71          // reset the color 
72          g.setColor(intialColor);
73      }
74      
75  
76  }
