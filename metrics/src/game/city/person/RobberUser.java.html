20      public void processInput(Input input) {
21  
22          boolean keyPressed = false; 
23          if (input.isKeyDown(Input.KEY_RIGHT)) {
24              moveRight();
25              
26              keyPressed = true; 
27          } else if (input.isKeyDown(Input.KEY_LEFT)) {
28              moveLeft();
29              keyPressed = true; 
30          }
31          if (input.isKeyDown(Input.KEY_UP)) {
32              moveUp();
33              keyPressed = true; 
34          } else if (input.isKeyDown(Input.KEY_DOWN)) {
35              moveDown();
36              keyPressed = true; 
37          }
38          if (input.isKeyDown(Input.KEY_SPACE)) {
39              rob();
40              keyPressed = true; 
41          } else if (input.isKeyPressed(Input.KEY_F)) {
42              if (nearByBldg != null) {
43                  nearByBldg.nextFlag();
44              }
45              keyPressed = true; 
46          }
47          else if(input.isKeyPressed(Input.KEY_W)) {
48              whistle();
49              keyPressed = true; 
50          }
51          if (!keyPressed)
52              stop();
53  
54      }

143     public boolean collides() {
144         // convert the position of the Player from pixels to 'Tile' position
145         // divide by the tile Size (in this case 16px)
146 
147         boolean isInCollision = false;
148         this.nearByBldg = null;
149 
150         // Check if trying to pass through gates
151         for (Gate gate : area.getGates()) {
152             if (gate.intersects(this.rect))
153                 return !gate.passThrough(this);
154         }
155 
156         // Collision with boundaries
157         Line[] boundLines = area.getMapBounds();
158         for (Line line : boundLines) {
159             if (line.intersects(this.rect))
160                 return true;
161         }
162 
163         // Collision wiht Buildings
164         for (Building bldg : area.getBuildings()) {
165             if (this.rect.intersects(bldg.getRect())) {
166                 this.nearByBldg = bldg;
167                 isInCollision = true;
168                 bldg.setShowBuildingInfo(true);
169                 break;
170             }
171             if (this.rect.intersects(bldg.robbingRegion)) {
172                 this.nearByBldg = bldg;
173                 bldg.setShowBuildingInfo(true);
174             }
175         }
176 //      final int x = (int)position.getX() / Globals.TILE_SIZE;
177 //      final int y = (int)position.getY() / Globals.TILE_SIZE;
178         return area.getCityMap().blocked(this) || isInCollision;
179     }
