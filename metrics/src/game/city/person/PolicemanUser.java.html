152     public void moveToNextStep() {
153 
154         final int period = 80; 
155         final int interval = Globals.TILE_SIZE; 
156 
157         final Timer timer = new Timer();
158         
159         final TimerTask timerTask = new TimerTask() {
160             int count = 0;  
161 
162             @Override
163             public void run() {
164 
165                 if (count==interval){
166                     
167                     count = 0; // reset count
168                     
169                     // make sure there are more steps to get
170                     if (pathToTargetIndex < pathToTarget.getLength() - 1){
171                         currentStep = nextStep;
172                         pathToTargetIndex++;
173                         nextStep = pathToTarget.getStep(pathToTargetIndex);
174                     }
175                     // if there are no more steps cancel the timer
176                     else
177                         timer.cancel();
178                 }
179                             
180                 int stepX = (nextStep.getX() - currentStep.getX()) * interval;
181                 int stepY = (nextStep.getY() - currentStep.getY()) * interval;              
182     
183                 // compute delta of prev and next steps
184                 float newX = position.getX() + stepX/interval; 
185                 float newY = position.getY() + stepY/interval; 
186 
187                 position.setX(newX);
188                 position.setY(newY);
189                 
190                 count++; 
191             }
192         };
193         timer.schedule(timerTask, 0, period/interval);
194 
195     }

314     @Override
315     public boolean collides() {
316         // convert the position of the Player from pixels to 'Tile' position
317         // divide by the tile Size (in this case 16px)
318 
319         
320         // Collision with boundaries
321         Line[] boundLines = area.getMapBounds();
322         for (Line line : boundLines) {
323             if (line.intersects(this.rect))
324                 return true;
325         }
326     
327         nearByBldg = null; 
328         boolean isInCollision = false;
329         for (Building bldg: area.getBuildings()) {
330             if (rect.intersects(bldg.getRect())) {
331                 isInCollision = true;
332                 nearByBldg = bldg; 
333                 bldg.setShowBuildingInfo(true);
334                 break;
335             }
336             if (this.rect.intersects(bldg.robbingRegion)) {
337                 this.nearByBldg = bldg;
338                 bldg.setShowBuildingInfo(true);
339             }
340         }
341         final int x = (int)position.getX() / Globals.TILE_SIZE;
342         final int y = (int)position.getY() / Globals.TILE_SIZE;
343         
344         return area.getCityMap().blocked(this) || isInCollision; 
345     }
