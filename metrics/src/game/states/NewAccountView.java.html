59      public void init(GameContainer arg0, StateBasedGame arg1)

65      @Override
66      public void render(GameContainer gc, StateBasedGame sbg, Graphics g)
67              throws SlickException {
68          super.render(gc, sbg, g);
69          g.drawImage(this.backgroundImage, 0, 0);
70          
71          g.drawString("Enter Username: " , TEXT_FIELD_X, TEXT_FIELD_Y-50);
72          usernameTextField.render(gc, g);
73  
74          if (createSuccess){
75              // save the previous graphics color
76              Color prevColor = g.getColor();
77  
78              // change the color
79              g.setColor(Color.red);
80  
81              // draw string
82              g.drawString("Failed to create account" , TEXT_FIELD_X, TEXT_FIELD_Y+50);
83  
84              // change the color back to what it was
85              g.setColor(prevColor);
86          }
87      }

89      @Override
90      public void enter(GameContainer container, StateBasedGame game)
91              throws SlickException {
92          super.enter(container, game);

94          createSuccess = false;

97      @Override
98      public void initButtons() {
99          GameContainer container = Game.getInstance().getContainer();
100 
101         // Font
102         TrueTypeFont font = Font.getFont(Globals.PETITINHO_FONT,16.0f);
103         
104         // Images
105         Image createAccountImage = null,
106                 backbuttonImage = null, 
107                 policeImage = null,
108                 robberImage = null;
109 
110         // Init Images
111         try{
112             createAccountImage = new Image(createAccountImagePath);
113             backbuttonImage = new Image(BACK_BUTTON_PATH);
114             robberImage = new Image(ROBBER_IMAGE_PATH);
115             policeImage = new Image(POLICE_IMAGE_PATH);
116         }
117         catch (SlickException se) {se.printStackTrace();}
118         
119         // Buttons  
120         MenuButton backButton = new MenuButton(container, backbuttonImage , 0, this.backgroundImage.getHeight()) {
121             @Override
122             public void performAction() {
123                 Game.getInstance().enterState(Globals.ACCOUNT_PICK);
124             }
125         };
126 
127         MenuButton robberButton = new MenuButton(container, robberImage, BUTTON_2_X, BUTTON_2_Y) {
128             @Override
129             public void performAction() {
130                 createAccount(true);
131             }
132         };
133         MenuButton policeButton = new MenuButton(container, policeImage, BUTTON_1_X, BUTTON_1_Y) {
134             @Override
135             public void performAction() {
136                 createAccount(false);
137             }
138         };
139         
140         // Text Field
141         usernameTextField = new TextField(container, font, TEXT_FIELD_X, TEXT_FIELD_Y,
142                 TEXT_FIELD_WIDTH, TEXT_FIELD_HEIGHT);
143         usernameTextField.setBackgroundColor(Color.white);
144         usernameTextField.setCursorVisible(true);
145         usernameTextField.setTextColor(Color.black);
146 
147 
148         // set the Buttons of the View
149         this.setButtons(robberButton, policeButton, backButton);
150     }
151 
152     public void createAccount(boolean isRobber){
153         AudioGame.playAsSound("button-21.ogg");
154 
155         // get Username from Textfield      
156         String username = usernameTextField.getText(); 
157 
158         // initialize account
159         Account account = new Account(username, isRobber);
160 
161         // create account 
162         createSuccess = account.create();
163 
164         if (createSuccess){
165             account.setIsRobber(isRobber);
166             Game.getInstance().enterState(Globals.ACCOUNT_PICK);
167         }       
168     }
