70      public void start() throws SlickException { 
71          /*
72           * When Resuming game, load the JSON Object and call the load method
73           */
74          if (isResumingGame) {
75              HashMap<Object, Object> resumeGame = 
76                      Game.getInstance()
77                      .getAccount().getResumeGame();
78              load(resumeGame);
79          }
80          else {
81  
82              /*
83               * Necessary for when the user plays several games, need to reset the fields of the class before starting
84               */
85              reset();
86  
87              area = new Area(level);     // Create Area
88          }
89  
90          userIsRobber = Game.getInstance().getAccount().getIsRobber();
91          setMainCharacter();
92          area.getCamera().setPerson(mainCharacter);
93          startGameTimer(); // Start timer
94      }

96      public void setMainCharacter() {
97          if (userIsRobber)
98              this.mainCharacter = getRobber();
99          else
100             this.mainCharacter = getPoliceOffice().getSelectedPoliceman();
101 
102         if (mainCharacter instanceof PolicemanUser) {
103             PolicemanUser police = (PolicemanUser) mainCharacter;
104             police.setSelected(true);
105         }
106         if (area!=null)
107             area.getCamera().setPerson(mainCharacter);
108     }

292     private Building selectBuilding(Point pnt) {
293         // create a rectangle and use the intersect method to check whether
294         // the policeman rect intersects with the mouse click
295         Rectangle rect = new Rectangle(area.getCamera().getCameraX() + pnt.getX()
296                 - Globals.SELECTION_ERROR / 2, area.getCamera().getCameraY() + pnt.getY()
297                 - Globals.SELECTION_ERROR / 2, Globals.SELECTION_ERROR,
298                 Globals.SELECTION_ERROR);
299 
300         for (Building bldg : area.getBuildings()) {
301             if (bldg.getRect().intersects(rect)) {
302                 return bldg;
303             }
304         }
305         return null;
306     }
307     private Road selectRoad(Point pnt) {
308         // create a rectangle and use the intersect method to check whether

312                 - Globals.SELECTION_ERROR / 2, Globals.SELECTION_ERROR,
313                 Globals.SELECTION_ERROR);
314 
315         for (Road road : area.getRoads()) {

321     }
322     /**
323      * @param destX
324      *            x-position of the mouse input
325      * @param destY
326      *            y-position of the mouse input
327      * @return a policeman object if the destX and destY passed are close to a
328      *         policeman in the map. Returns null otherwise
329      */
330     private Policeman selectPoliceman(int destX, int destY) {
331         // create a rectangle and use the intersect method to check whether
332         // the policeman rect intersects with the mouse click
333         Rectangle rect = new Rectangle(area.getCamera().getCameraX() + destX
334                 - Globals.SELECTION_ERROR / 2, area.getCamera().getCameraY() + destY
335                 - Globals.SELECTION_ERROR / 2, Globals.SELECTION_ERROR,
336                 Globals.SELECTION_ERROR);
337 
338         for (Policeman policeman : area.getPoliceOffice().getPoliceForceArray()) {
339             if (policeman.rect.intersects(rect)) {
340                 return policeman;
341             }
342         }
343         return null;
344     }

364     public void gameOver(boolean youWin) {
365 
366         if (isGameOver)
367             return; 
368         
369         Account account = Game.getInstance().getAccount(); 
370 
371         
372         if (youWin){
373             AudioGame.playAsSound("success.ogg");
374             account.setHighestLevelReached(area.getLevel()+1);
375         }
376         account.saveScore(getGameTime(), getScore(), youWin, userIsRobber);
377         
378         
379         GameOverState gameOver = (GameOverState) Game.getInstance().getState(Globals.GAME_OVER);
380         gameOver.set(userIsRobber, youWin);
381 
382         Game.getInstance().enterState(Globals.GAME_OVER,
383                 new FadeOutTransition(), new FadeInTransition());
384 
385 
386         isGameOver = true;
387     }

461     @Override
462     public void load(Object loadObj) {
463 
464         HashMap<Object, Object> resumeGame = (HashMap<Object, Object>) loadObj;
465 
466         assert resumeGame != null : "There is some type of inconsistency in the resume game";
467 
468         level = (int) resumeGame.get(Globals.LEVEL);
469         gameTime = (int) resumeGame.get(Globals.TIME);
470 
471         // LOAD: Area
472         Object areaObj = resumeGame.get(Globals.AREA);
473         area = new Area(level);
474         area.load(areaObj);
475 
476 
477         setResumingGame(false);
478     }
