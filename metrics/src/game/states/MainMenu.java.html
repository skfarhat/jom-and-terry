10  import org.newdawn.slick.GameContainer;

13  import org.newdawn.slick.SlickException;
14  import org.newdawn.slick.state.StateBasedGame;

16  /**
17   * Inherits from the abstract class Menu
18   * @author Sami
19   *
20   */
21  public class MainMenu extends Menu
22  {
23  
24      private StateBasedGame gameContainer = null;
25  
26      private final String BACK_BUTTON_PATH               = "res/buttons/back-button.png";
27      private final String RESUME_GAME_IMAGEPATH          = "res/buttons/resume-game-button.png";
28      private final String RESUME_GAME_DISABLED_IMAGEPATH = "res/buttons/resume-game-disabled.png";
29      private final String NEW_GAME_IMAGEPATH             = "res/buttons/new-game-button.png";    
30      private final String SETTINGS_IMAGEPATH             = "res/buttons/settings-button.png";
31      private final String STATISTICS_IMAGEPATH           = "res/buttons/statistics-button.png";
32      private final String EXIT_IMAGEPATH                 = "res/buttons/exit-button.png";
33      private final String BACKGROUND_IMAGEPATH           = "res/background.png";
34  
35      private static Integer backgroundX; 
36      private Image backgroundImage = null;
37  
38      private Account account = null; 
39  
40      public MainMenu(int state, StateBasedGame sbg){
41          super();
42          this.gameContainer = sbg; 
43      }
44  
45      @Override
46      public void init(GameContainer arg0, StateBasedGame arg1)
47              throws SlickException {
48  
49          this.backgroundImage = new Image(BACKGROUND_IMAGEPATH);
50          backgroundX = (Globals.APP_WIDTH-backgroundImage.getWidth())/2;
51  
52      }
53  
54      @Override
55      public void render(GameContainer gc, StateBasedGame sbg, Graphics g)
56              throws SlickException {
57  
58          g.drawImage(this.backgroundImage, backgroundX, 0);
59          for(MenuButton b : buttons)
60          {
61              b.render(g);
62          }
63      }
64      
65      @Override
66      public void enter(GameContainer container, StateBasedGame game)
67              throws SlickException {
68          super.enter(container, game);
69          initButtons();
70          Play.getInstance().reset(); 
71      }
72  
73      @Override
74      public void initButtons() {
75          Image exitImage = null, statisticsImage = null, settingsImage = null, backButtonImage = null, newGameButtonImage = null, 
76                  resumeGameButtonImage = null, resumeGameButtonDisabledImage = null; 
77  
78          try{
79  
80              newGameButtonImage  = new Image(NEW_GAME_IMAGEPATH);                        // NEW GAME
81              resumeGameButtonImage  = new Image(RESUME_GAME_IMAGEPATH);                  // RESUME GAME
82              resumeGameButtonDisabledImage  = new Image(RESUME_GAME_DISABLED_IMAGEPATH); // RESUME GAME DISABLED
83              backButtonImage = new Image(BACK_BUTTON_PATH);                              // BACK 
84              settingsImage = new Image(SETTINGS_IMAGEPATH);                              // SETTINGS
85              statisticsImage = new Image(STATISTICS_IMAGEPATH);                          // STATISTICS
86              exitImage = new Image(EXIT_IMAGEPATH);                                      // EXIT
87          }   
88          catch (SlickException se)
89          {
90              se.printStackTrace();
91          }
92  
93          final float height = 50.0f; 
94          final int verticalMargin = 70; 
95  
96          final int x1 = (int) (gameContainer.getContainer().getWidth()/2.0f - resumeGameButtonImage.getWidth()/2.0f); 
97          final int y1 = (int) (gameContainer.getContainer().getHeight()/2.0f - height/2.0f);
98  
99          final int x2 = (int) (gameContainer.getContainer().getWidth()/2.0f - newGameButtonImage.getWidth()/2.0f); 
100         final int y2 = (int) y1 + verticalMargin*1;
101 
102         final int x3 = (int) (gameContainer.getContainer().getWidth()/2.0f - statisticsImage.getWidth()/2.0f); 
103         final int y3 = (int) y1 + verticalMargin*2;
104 
105         final int x4 = (int) (gameContainer.getContainer().getWidth()/2.0f - settingsImage.getWidth()/2.0f); 
106         final int y4 = (int) y1 + verticalMargin*3;
107 
108         final int x5 = (int) (gameContainer.getContainer().getWidth()/2.0f - exitImage.getWidth()/2.0f); 
109         final int y5 = (int) y1 + verticalMargin*4;
110 
111 
112 
113         final boolean canResume = Game.getInstance().getAccount().getResumeGame()!=null; 
114         Image resumeGameImage = (canResume)? resumeGameButtonImage : resumeGameButtonDisabledImage;
115         
116 
117         MenuButton backButton = new MenuButton(this.gameContainer.getContainer(), backButtonImage, 0, this.backgroundImage.getHeight()) {
118             @Override
119             public void performAction() {
120                 gameContainer.enterState(Globals.ACCOUNT_PICK);
121             }
122         };
123         MenuButton resumeGameButton = new MenuButton(this.gameContainer.getContainer(), resumeGameImage, x1, y1) {
124             @Override
125             public void performAction() {
126                 AudioGame.playAsSound("button-21.ogg");
127                 Play.getInstance().setResumingGame(true);
128                 gameContainer.enterState(Globals.PLAY);
129             }
130         };
131         resumeGameButton.setEnabled(canResume);
132 
133         MenuButton newGameButton = new MenuButton(this.gameContainer.getContainer(), newGameButtonImage, x2, y2) {
134             @Override
135             public void performAction() {
136                 AudioGame.playAsSound("button-21.ogg");
137                 if (canResume)
138                 {
139                     Game.getInstance().getAccount().setResumeGame(null);
140                     Game.getInstance().getAccount().removeResumeGame();
141                 }
142                 gameContainer.enterState(Globals.AREA_PICK);
143             }
144         };
145         MenuButton statisticsButton = new MenuButton(this.gameContainer.getContainer(), statisticsImage, x3, y3) {
146             @Override
147             public void performAction() {
148                 AudioGame.playAsSound("button-21.ogg");
149                 
150                 // get the stats view
151                 StatsView statsView = (StatsView) gameContainer.getState(Globals.STATS_VIEW);
152 
153                 // set the account
154                 statsView.setAccount(account);
155 
156                 // enter the new view (state) 
157                 gameContainer.enterState(Globals.STATS_VIEW);
158             }
159         };
160         MenuButton settingsButton = new MenuButton(this.gameContainer.getContainer(), settingsImage, x4, y4) {
161             @Override
162             public void performAction() {
163                 AudioGame.playAsSound("button-21.ogg");
164                 gameContainer.enterState(3);
165 
166             }
167         };
168         MenuButton exitButton = new MenuButton(this.gameContainer.getContainer(), exitImage, x5, y5) {
169             @Override
170             public void performAction() {
171                 AudioGame.playAsSound("button-21.ogg");
172                 gameContainer.getContainer().exit();
173             }
174         };
175 
176         this.setButtons(resumeGameButton, newGameButton, backButton,settingsButton, statisticsButton, exitButton); 
177     }
178 
179     @Override
180     public int getID() {
181         return Globals.MAIN_MENU;
182     }
183 
184 }
