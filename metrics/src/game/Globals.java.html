142     public static final float distance (Point pnt1, Point pnt2){
143         float distance = (float) Math.sqrt(Math.pow(
144                 pnt1.getX() - pnt2.getX(), 2.0)
145                 + Math.pow(pnt1.getY() - pnt2.getY(), 2.0)
146                 );
147         return distance; 
148     }
